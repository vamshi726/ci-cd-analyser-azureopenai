const API_BASE_URL = import.meta.env.VITE_API_URL || 'http://localhost:8000';

  baseURL: API_BASE_URL,
  headers: {
    'Content-Type': 'application/json',
  },

export const api = {
  // Health check
  health: () => apiClient.get('/health'),

  // GitLab
  getLatestPipeline: () => apiClient.get('/api/latest-pipeline'),
  getLatestPipelineLogs: () => apiClient.get('/api/latest-pipeline-logs'),

  // Analysis
  analyzeLatest: () => apiClient.post('/api/analyze-latest'),
  analyzeFailure: (data: any) => apiClient.post('/api/analyze', data),

  // Failures
  getFailures: (params?: { category?: string; limit?: number }) =>
    apiClient.get('/api/failures', { params }),
  getFailureDetail: (id: string) => apiClient.get(`/api/failures/${id}`),

  // Metrics
  getMetricsSummary: () => apiClient.get('/api/metrics/summary'),
};

src/components/ConfidenceBar.tsx
typescript
import React from 'react';

interface ConfidenceBarProps {
  confidence: number;
}

export default function ConfidenceBar({ confidence }: ConfidenceBarProps) {
  const percentage = Math.round(confidence * 100);
  
  const getColor = (conf: number) => {
    if (conf >= 0.8) return 'bg-green-500';
    if (conf >= 0.6) return 'bg-yellow-500';
    return 'bg-orange-500';
  };

  return (
    <div className="w-full">
      <div className="flex items-center justify-between mb-2">
        <span className="text-sm font-medium text-gray-700">Confidence</span>
        <span className="text-sm font-bold text-gray-900">{percentage}%</span>
      </div>
      <div className="w-full bg-gray-200 rounded-full h-3">
        <div
          className={`h-3 rounded-full transition-all duration-300 ${getColor(confidence)}`}
          style={{ width: `${percentage}%` }}
        />
      </div>
    </div>
  );
}

import React from 'react';
import { useQuery } from '@tanstack/react-query';
import { AlertCircle, Clock, TrendingDown, BarChart3 } from 'lucide-react';
import { api } from '../api/client';
import MetricCard from '../components/MetricCard';
import { MetricsSummary, Failure } from '../types';

export default function Dashboard() {
  const { data: metrics } = useQuery<MetricsSummary>({
    queryKey: ['metrics'],
    queryFn: () => api.getMetricsSummary().then(r => r.data),
    refetchInterval: 30000, // Refresh every 30 seconds
  });

  const { data: failures } = useQuery<Failure[]>({
    queryKey: ['failures'],
    queryFn: () => api.getFailures({ limit: 10 }).then(r => r.data),
    refetchInterval: 30000,
  });

  const avgTimeMinutes = metrics 
    ? (metrics.avg_processing_time_ms / 1000 / 60).toFixed(1) 
    : '0';
  
  const timeSaved = metrics 
    ? ((45 - parseFloat(avgTimeMinutes)) * metrics.total_failures / 60).toFixed(1) 
    : '0';

  const categoryCount = metrics 
    ? Object.keys(metrics.category_breakdown).length 
    : 0;

  return (
    <div className="space-y-8">
      {/* Header */}
      <div>
        <h1 className="text-3xl font-bold text-gray-900">Dashboard</h1>
        <p className="text-gray-600 mt-2">Real-time CI/CD failure analysis metrics</p>
      </div>

      {/* KPI Cards */}
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6">
        <MetricCard
          title="Total Failures"
          value={metrics?.total_failures || 0}
          icon={AlertCircle}
          iconColor="bg-red-500"
        />
        <MetricCard
          title="Avg RCA Time"
          value={`${avgTimeMinutes}m`}
          icon={Clock}
          iconColor="bg-blue-500"
          trend="vs 45m manual"
        />
        <MetricCard
          title="Time Saved"
          value={`${timeSaved}h`}
          icon={TrendingDown}
          iconColor="bg-green-500"
          trend="per team/day"
        />
        <MetricCard
          title="Categories"
          value={categoryCount}
          icon={BarChart3}
          iconColor="bg-purple-500"
        />
      </div>

      {/* Category Breakdown */}
      <div className="bg-white p-6 rounded-lg shadow">
        <h2 className="text-xl font-bold text-gray-900 mb-4">Failure Categories</h2>
        <div className="space-y-3">
          {metrics && Object.entries(metrics.category_breakdown).map(([category, count]) => (
            <div key={category} className="flex items-center justify-between">
              <span className="text-gray-700 font-medium">{category}</span>
              <div className="flex items-center gap-3">
                <div className="w-48 bg-gray-200 rounded-full h-3">
                  <div
                    className="bg-blue-600 h-3 rounded-full transition-all"
                    style={{ 
                      width: `${(count / metrics.total_failures) * 100}%` 
                    }}
                  />
                </div>
                <span className="text-gray-900 font-bold w-12 text-right">
                  {count}
                </span>
              </div>
            </div>
          ))}
        </div>
      </div>

      {/* Recent Failures */}
      <div className="bg-white p-6 rounded-lg shadow">
        <h2 className="text-xl font-bold text-gray-900 mb-4">Recent Failures</h2>
        <div className="space-y-3">
          {failures?.slice(0, 5).map((failure) => (
            <div 
              key={failure.id} 
              className="flex items-center justify-between p-3 bg-gray-50 rounded hover:bg-gray-100 transition-colors cursor-pointer"
              onClick={() => window.location.href = `/failures/${failure.failure_id}`}
            >
              <div>
                <p className="font-medium text-gray-900">{failure.job_name}</p>
                <p className="text-sm text-gray-600">{failure.error_type}</p>
              </div>
              <span className={`px-3 py-1 rounded text-sm font-medium ${
                failure.failure_category === 'Test' ? 'bg-yellow-100 text-yellow-800' :
                failure.failure_category === 'Infrastructure' ? 'bg-red-100 text-red-800' :
                failure.failure_category === 'Auth' ? 'bg-orange-100 text-orange-800' :
                'bg-blue-100 text-blue-800'
              }`}>
                {failure.failure_category}
              </span>
            </div>
          ))}
        </div>
      </div>
    </div>
  );
}

import React from 'react';
import { useParams, useNavigate } from 'react-router-dom';
import { useQuery } from '@tanstack/react-query';
import { ArrowLeft, Clock, AlertCircle, CheckCircle, TrendingUp } from 'lucide-react';
import { api } from '../api/client';
import CategoryBadge from '../components/CategoryBadge';
import ConfidenceBar from '../components/ConfidenceBar';
import { Failure } from '../types';

export default function FailureDetail() {
  const { id } = useParams<{ id: string }>();
  const navigate = useNavigate();

  const { data: failure, isLoading } = useQuery<Failure>({
    queryKey: ['failure', id],
    queryFn: () => api.getFailureDetail(id!).then(r => r.data),
    enabled: !!id,
  });

  if (isLoading) {
    return (
      <div className="flex items-center justify-center h-96">
        <div className="text-center">
          <div className="inline-block animate-spin rounded-full h-12 w-12 border-b-2 border-blue-600"></div>
          <p className="mt-4 text-gray-600">Loading failure details...</p>
        </div>
      </div>
    );
  }

  if (!failure) {
    return (
      <div className="text-center py-12">
        <p className="text-gray-600">Failure not found</p>
      </div>
    );
  }

  return (
    <div className="space-y-6">
      {/* Back Button */}
      <button
        onClick={() => navigate('/failures')}
        className="flex items-center gap-2 text-blue-600 hover:text-blue-700"
      >
        <ArrowLeft className="w-4 h-4" />
        Back to Failures
      </button>

      {/* Header Card */}
      <div className="bg-white p-6 rounded-lg shadow">
        <div className="flex items-start justify-between">
          <div>
            <h1 className="text-2xl font-bold text-gray-900">{failure.job_name}</h1>
            <p className="text-gray-600 mt-1">
              Pipeline #{failure.pipeline_id} ‚Ä¢ Stage: {failure.stage}
            </p>
          </div>
          <CategoryBadge category={failure.failure_category} />
        </div>
      </div>

      {/* Error Type */}
      <div className="bg-red-50 border-l-4 border-red-500 p-6 rounded-lg">
        <div className="flex items-start gap-3">
          <AlertCircle className="w-6 h-6 text-red-600 flex-shrink-0 mt-0.5" />
          <div>
            <h3 className="text-lg font-semibold text-red-900">
              {failure.error_type}
            </h3>
            <p className="text-red-700 mt-2">{failure.root_cause}</p>
          </div>
        </div>
      </div>

      {/* Suggested Fix */}
      <div className="bg-green-50 border-l-4 border-green-500 p-6 rounded-lg">
        <div className="flex items-start gap-3">
          <CheckCircle className="w-6 h-6 text-green-600 flex-shrink-0 mt-0.5" />
          <div className="flex-1">
            <h3 className="text-lg font-semibold text-green-900 mb-3">
              Suggested Fix
            </h3>
            <p className="text-green-800 mb-4">{failure.suggested_fix}</p>
            
            {failure.fix_commands && failure.fix_commands.length > 0 && (
              <div className="bg-gray-900 text-gray-100 p-4 rounded font-mono text-sm space-y-2">
                {failure.fix_commands.map((cmd, idx) => (
                  <div key={idx} className="flex items-start gap-2">
                    <span className="text-gray-500">$</span>
                    <span>{cmd}</span>
                  </div>
                ))}
              </div>
            )}
          </div>
        </div>
      </div>

      {/* Confidence */}
      <div className="bg-white p-6 rounded-lg shadow">
        <ConfidenceBar confidence={failure.confidence} />
      </div>

            <h3 className="text-lg font-semibold text-blue-900">

src/App.tsx
typescript
import React from 'react';
import { BrowserRouter, Routes, Route, Link, useLocation } from 'react-router-dom';
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { Home, AlertCircle, TrendingDown, Activity } from 'lucide-react';
import Dashboard from './pages/Dashboard';
import FailureList from './pages/FailureList';
import FailureDetail from './pages/FailureDetail';
import BeforeAfter from './pages/BeforeAfter';

const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      refetchOnWindowFocus: false,
      retry: 1,
    },
  },
});

function Navigation() {
  const location = useLocation();

  const navItems = [
    { path: '/', label: 'Dashboard', icon: Home },
    { path: '/failures', label: 'Failures', icon: AlertCircle },
    { path: '/compare', label: 'Before/After', icon: TrendingDown },
  ];

  return (
    <nav className="bg-blue-900 text-white shadow-lg">
      <div className="max-w-7xl mx-auto px-4 py-4">
        <div className="flex items-center justify-between">
          <div className="flex items-center gap-2">
            <Activity className="w-8 h-8" />
            <h1 className="text-2xl font-bold">CI/CD RCA System</h1>
          </div>
          <div className="flex gap-6">
            {navItems.map(({ path, label, icon: Icon }) => (
              <Link
                key={path}
                to={path}
                className={`flex items-center gap-2 px-3 py-2 rounded transition-colors ${
                  location.pathname === path
                    ? 'bg-blue-800 text-white'
                    : 'text-blue-100 hover:text-white hover:bg-blue-800'
                }`}
              >
                <Icon className="w-4 h-4" />
                {label}
              </Link>
            ))}
          </div>
        </div>
      </div>
    </nav>
  );
}

function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <BrowserRouter>
        <div className="min-h-screen bg-gray-50">
          <Navigation />
          <main className="max-w-7xl mx-auto px-4 py-8">
            <Routes>
              <Route path="/" element={<Dashboard />} />
              <Route path="/failures" element={<FailureList />} />
              <Route path="/failures/:id" element={<FailureDetail />} />
              <Route path="/compare" element={<BeforeAfter />} />
            </Routes>
          </main>
        </div>
      </BrowserRouter>
    </QueryClientProvider>
  );
}

export default App;
